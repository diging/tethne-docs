[docs]    def test_transform(self):
        corpus = read(datapath)
        wordcounts = corpus.features['wordcounts']
        def filter(f, v, c, dc):
            if f == 'the':
                return 0
            return v

        # filter() should remove a single token.
        self.assertEqual(len(wordcounts.index) - 1,
                         len(wordcounts.transform(filter).index))
